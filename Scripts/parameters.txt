1-build.sh <cpu_arch> <build_preset> <build_type>
	-> Build-$OS_TYPE-$CPU_ARCH-$BUILD_PRESET-$BUILD_TYPE
	CPU_ARCH      -> build dir, build tools
	BUILD_PRESET  -> build dir, build config
	BUILD_TYPE    -> build dir, build config

1-build.bat <target_env> <linkage> <build_type>
	-> Build-$OS_TYPE-$TARGET_ENV-$LINKAGE-$BUILD_TYPE
	TARGET_ENV    -> build dir, build tools
	LINKAGE       -> build dir, build tools
	BUILD_TYPE    -> build dir, build config

2-package.sh <build_dir> <package_type> <os_type> <cpu_arch>
	-> $PROJECT_NAME-$APP_VERSION-$OS_TYPE-$CPU_ARCH.AppImage
	OS_TYPE       -> package name, verify package type, executable path
	CPU_ARCH      -> package name, package tools
	PACKAGE_TYPE  -> packaging procedure

2-package.bat <build_dir> <target_env> <linkage> <build_type>
	-> $PROJECT_NAME-$APP_VERSION-$OS_TYPE-$TARGET_ENV-$LINKAGE.zip
	TARGET_ENV    -> package name
	LINKAGE       -> package name
	BUILD_TYPE    -> executable path
